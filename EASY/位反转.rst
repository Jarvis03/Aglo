    unsigned int reverse(unsigned int x)
    {
        x = ((x & 0xaaaaaaaa) >> 1) | ((x & 0x55555555) << 1);
        x = ((x & 0xcccccccc) >> 1) | ((x & 0x33333333) << 1);
        x = ((x & 0xf0f0f0f0) >> 1) | ((x & 0x0f0f0f0f) << 1);
        x = ((x & 0xff00ff00) >> 1) | ((x & 0x00ff00ff) << 1);
        return ((x >> 16) | (x << 16));
    }
